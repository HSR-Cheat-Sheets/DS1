\subsection{Virtual Machine}
\begin{itemize}[label={\textcolor{ForestGreen}{+}}]
    \item App can access all OS resources
    \item Live migrations
\end{itemize}
\begin{itemize}[label={\textcolor{ForestGreen}{+}/\textcolor{red}{--}}]
    \item Pre allocates memory
    \item Full isolation
\end{itemize}

\subsection{Container}
\begin{itemize}[label={\textcolor{ForestGreen}{+}}]
    \item Reduced IT management resources
    \item Reduced size of snapshots
    \item Quicker spinning up apps
\end{itemize}
\begin{itemize}[label={\textcolor{ForestGreen}{+}/\textcolor{red}{--}}]
    \item Available memory is shared
    \item Process-based isolation (share same kernel)
\end{itemize}
